rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Allow authenticated users to upload and read chat files
    match /chat_files/{appointmentId}/{fileName} {
      allow read, write: if request.auth != null;
    }
    
    // Allow authenticated users to upload profile images
    match /profile_images/{userId}/{fileName} {
      allow read, write: if request.auth != null;
    }
    
    // Allow authenticated users to upload appointment related files
    match /appointment_files/{appointmentId}/{fileName} {
      allow read, write: if request.auth != null;
    }
    
    // Allow authenticated users to upload prescription files
    match /prescription_files/{appointmentId}/{fileName} {
      allow read, write: if request.auth != null;
    }
    
    // Default rule - allow all authenticated users (covers any other paths)
    match /{allPaths=**} {
      allow read, write: if request.auth != null;
    }
  }
}
